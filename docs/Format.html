    Format

    (abstract) new Format()

    Transforms ConfigFile data after being read and content before being written
    Source:
    • Format.js, line 45
    Example
    import {
     Format,
     FormatReturnObject
} from "@aery/mlc";

class MyFormat extends Format {

     read(data: Buffer, default_content?: string, default_options?: null): FormatReturnObject {
         // Do something with data
         if (data) {
             return {
                 content: data.toString(),
                 defaulted: false
             };
         } else {
             if (default_content) {
                 return {
                     content: default_content,
                     defaulted: true
                 };
             } else {
                 return {
                     content: data,
                     defaulted: false
                 };
             }
         }
     }

     write(content: string): string {
         // Do something with content
         return content;
     }

}
